#include 
#include "3dengfx_config.h"
#include "3dengfx/3dengfx.hpp"
#include "common/timer.h"
#include "fxwt/fxwt.hpp"

bool Init();
void CleanUp();
void UpdateGfx();
void KeyHandler(int key);

int main() {
    if(!Init()) {
        return -1;
    }

    return fxwt::MainLoop();
}

bool Init() {
    try {
        GraphicsInitParameters gip = LoadGraphicsContextConfig("3dengfx.conf");
        CreateGraphicsContext(gip);
    }
    catch(EngineException e) {
        std::cerr << e.GetMessage() << std::endl;
        return false;
    }

    fxwt::SetDisplayHandler(UpdateGfx);
    fxwt::SetIdleHandler(UpdateGfx);
    fxwt::SetKeyboardHandler(KeyHandler);
    atexit(CleanUp);

    return true;
}

void UpdateGfx() {
    Clear(Color(0.0f, 0.0f, 1.0f));
  
    Flip();
}

void CleanUp() {
    DestroyGraphicsContext();
}

void KeyHandler(int key) {
    if(key == 27) exit(0);
}
